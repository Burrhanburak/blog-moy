import Accordion from "@/components/mintlify/Accordion";

  DollarSign,
  Calendar,
  Zap,
  BookOpen,
  Home,

  Megaphone,
} from "lucide-react";

export default function CityCategoryTemplate({
  city,
  state,
  country,
  category,
  year,
  intro,
  keyTakeaways = [],
  providers = [],
  pricing,
  pricingRange,
  pricingNotes = [],

  compareLinks = [],
  nearbyAreas = [],
  relatedPosts = [],
  cta,

  sections = [],
  topicCluster,

}) {
  const linkClass =
    "underline decoration-dotted underline-offset-4 text-[#ff4d00] hover:text-blue-700";

const slugify = (value) =>
(value || "")
.toLowerCase()
.normalize("NFD")
.replace(/[\u0300-\u036f]/g, "")
.replace(/[^a-z0-9]+/g, "-")
.replace(/(^-|-$)+/g, "");

const countrySlug = country?.slug || slugify(country?.name);
const stateSlug = state?.slug || slugify(state?.name);
const citySlug = city?.slug || slugify(city?.name);

const makeCityCategoryPath = (categoryKey) => {
const segments = [countrySlug, stateSlug, citySlug].filter(Boolean);
if (categoryKey) {
segments.push(categoryKey);
}
return segments.length ? `/${segments.join("/")}` : "/";
};

const statePath =
countrySlug && stateSlug ? `/${countrySlug}/${stateSlug}` : undefined;

const cityCategoryHref = makeCityCategoryPath(category.key);
const citySeoHref = makeCityCategoryPath("seo-services");
const cityShopifyHref = makeCityCategoryPath("shopify-development");
const cityWordpressHref = makeCityCategoryPath("wordpress-development");
const cityDigitalMarketingHref = makeCityCategoryPath("digital-marketing");
const cityEcommerceHref = makeCityCategoryPath("ecommerce-development");
const cityAppHref = makeCityCategoryPath("app-development");
const citySocialHref = makeCityCategoryPath("social-media-marketing");

const cityCrmHref = makeCityCategoryPath("crm-development");
const cityErpHref = makeCityCategoryPath("erp-software");

const imageFallbacks = [
dynamicImage,

{ label: `${state.name} social media marketing`, href: statePath || citySocialHref },

{

},
{

year over year.`, ];

? faqs

{

})),
